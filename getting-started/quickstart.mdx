## 1. Install the SDK package

<CodeGroup>

```bash Next.js
pnpm add @wacht/nextjs @wacht/jsx @wacht/types
```

```bash React Router
pnpm add @wacht/react-router @wacht/jsx @wacht/types react-router
```

```bash TanStack Router
pnpm add @wacht/tanstack-router @wacht/jsx @wacht/types @tanstack/react-router
```

```bash Rust
cargo add wacht-rs
cargo add tokio --features full
```

</CodeGroup>

## 2. Add environment variables

<CodeGroup>

```bash Next.js
NEXT_PUBLIC_WACHT_PUBLISHABLE_KEY=pk_test_xxx
# Optional, only for backend API calls from server code
WACHT_API_KEY=wk_live_xxx
```

```bash React Router / TanStack Router
# Browser
VITE_WACHT_PUBLISHABLE_KEY=pk_test_xxx

# Server helpers
WACHT_PUBLISHABLE_KEY=pk_test_xxx
# Optional, only for backend API calls from server code
WACHT_API_KEY=wk_live_xxx
```

```bash Rust
WACHT_API_KEY=wk_live_xxx
# Optional if you do not use WACHT_PUBLISHABLE_KEY fallback
WACHT_FRONTEND_API_URL=https://<deployment>.fapi.trywacht.xyz
```

</CodeGroup>

## 3. Mount provider in your app

<CodeGroup>

```tsx Next.js (app/layout.tsx)
import { DeploymentProvider } from '@wacht/nextjs'

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html>
      <body>
        <DeploymentProvider publicKey={process.env.NEXT_PUBLIC_WACHT_PUBLISHABLE_KEY!}>
          {children}
        </DeploymentProvider>
      </body>
    </html>
  )
}
```

```tsx React Router
import { BrowserRouter } from 'react-router'
import { DeploymentProvider } from '@wacht/react-router'

export function App() {
  return (
    <DeploymentProvider publicKey={import.meta.env.VITE_WACHT_PUBLISHABLE_KEY}>
      <BrowserRouter>{/* routes */}</BrowserRouter>
    </DeploymentProvider>
  )
}
```

```tsx TanStack Router
import { createRouter, RouterProvider } from '@tanstack/react-router'
import { DeploymentProvider } from '@wacht/tanstack-router'
import { routeTree } from './routeTree.gen'

const router = createRouter({ routeTree })

export function App() {
  return (
    <DeploymentProvider publicKey={import.meta.env.VITE_WACHT_PUBLISHABLE_KEY}>
      <RouterProvider router={router} />
    </DeploymentProvider>
  )
}
```

</CodeGroup>

## 4. Protect a server-side request

<CodeGroup>

```ts Next.js (proxy.ts)
import { NextResponse } from 'next/server'
import { createRouteMatcher, wachtMiddleware } from '@wacht/nextjs/server'

const isProtected = createRouteMatcher(['/dashboard(.*)', '/api/private(.*)'])

export default wachtMiddleware(async (auth, req) => {
  if (!isProtected(req)) return NextResponse.next()
  await auth.protect()
  return NextResponse.next()
})
```

```ts React Router (loader)
import { json, redirect, type LoaderFunctionArgs } from 'react-router'
import { authenticateRequest } from '@wacht/react-router/server'

export async function loader({ request }: LoaderFunctionArgs) {
  const { auth, headers } = await authenticateRequest(request, {
    publishableKey: process.env.WACHT_PUBLISHABLE_KEY,
  })

  if (!auth.userId) throw redirect('/sign-in', { headers })
  return json({ userId: auth.userId }, { headers })
}
```

```ts TanStack Router (server function/loader)
import { authenticateRequest } from '@wacht/tanstack-router/server'

export async function getDashboardData(request: Request) {
  const { auth, headers } = await authenticateRequest(request, {
    publishableKey: process.env.WACHT_PUBLISHABLE_KEY,
  })

  if (!auth.userId) {
    return new Response(null, {
      status: 302,
      headers: {
        ...Object.fromEntries(headers.entries()),
        Location: '/sign-in',
      },
    })
  }

  return Response.json({ userId: auth.userId }, { headers })
}
```

```rust Rust (service client)
use wacht_rs::WachtClient;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let client = WachtClient::from_env()?;
    let users = client.users().list_users(10, 0).await?;
    println!("users: {}", users.data.len());
    Ok(())
}
```
