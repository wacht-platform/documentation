# DeploymentInitializing

Renders its children only while the deployment configuration is loading. Returns `null` once loaded.

```tsx
import { DeploymentInitializing } from "@wacht/tanstack-router";

export default function App() {
  return (
    <DeploymentInitializing>
      <LoadingSpinner />
    </DeploymentInitializing>
  );
}
```

## Loading Screen

```tsx
import { DeploymentInitializing, DeploymentInitialized } from "@wacht/tanstack-router";

export default function Root() {
  return (
    <>
      <DeploymentInitializing>
        <div className="loading-screen">
          <Spinner />
          <p>Loading...</p>
        </div>
      </DeploymentInitializing>
      <DeploymentInitialized>
        <App />
      </DeploymentInitialized>
    </>
  );
}
```

## Params

<ParamField path="children" type="ReactNode">
Content to render during loading.
</ParamField>
