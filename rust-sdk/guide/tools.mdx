# AI Tools API Guide

Use `client.ai().tools()` for backend AI tool management.

## List tools

```rust
use wacht::{WachtClient, WachtConfig};

let client = WachtClient::new(WachtConfig::new("wk_live_...", "https://app.wacht.dev"))?;

let tools = client
    .ai()
    .tools()
    .list_tools()
    .limit(50)
    .offset(0)
    .is_active(true)
    .search("weather")
    .send()
    .await?;

for tool in tools.data {
    println!("{}", tool.name);
}
```

## Get a tool

```rust
let tool = client
    .ai()
    .tools()
    .fetch_tool("tool_123")
    .send()
    .await?;

println!("{}", tool.name);
```

## Create a tool

```rust
use wacht::models::{
    AiToolConfiguration, ApiToolConfiguration, CreateAiToolRequest, HttpMethod,
};

let tool = client
    .ai()
    .tools()
    .create_tool(CreateAiToolRequest {
        name: "weather-api".to_string(),
        description: Some("Fetch weather information".to_string()),
        tool_type: "api".to_string(),
        requires_user_approval: false,
        configuration: AiToolConfiguration::Api(ApiToolConfiguration {
            endpoint: "https://api.weather.com".to_string(),
            method: HttpMethod::Get,
            authorization: None,
            request_body_schema: None,
            url_params_schema: None,
            timeout_seconds: Some(30),
        }),
    })
    .send()
    .await?;
```

## Update and delete

```rust
use wacht::models::UpdateAiToolRequest;

let updated = client
    .ai()
    .tools()
    .update_tool(
        "tool_123",
        UpdateAiToolRequest {
            name: Some("weather-api-v2".to_string()),
            ..Default::default()
        },
    )
    .send()
    .await?;

client.ai().tools().delete_tool("tool_123").send().await?;
```

## Agent tool attachments

```rust
client
    .ai()
    .tools()
    .attach_tool("agent_123", "tool_123")
    .send()
    .await?;

let attached = client
    .ai()
    .tools()
    .list_agent_tools("agent_123")
    .send()
    .await?;
```
